SELECT Code, Name, PlacedBy, FoundByMeDate FROM Caches WHERE UPPER(Name) LIKE "%_BALL_%" or UPPER(Name) LIKE "%_BASEBALL_%" or UPPER(Name) LIKE "%_COACH_%" or UPPER(Name) LIKE "%_FIELD_%" or UPPER(Name) LIKE "%_FIRST_%" or UPPER(Name) LIKE "%_HOME_%" or UPPER(Name) LIKE "%_INNING_%" or UPPER(Name) LIKE "%_PLAYER_%" or UPPER(Name) LIKE "%_SECOND_%" or UPPER(Name) LIKE "%_STRIKE_%" or UPPER(Name) LIKE "%_THIRD_%" or UPPER(Name) LIKE "%_WALK_%" ORDER BY FoundByMeDate DESC
